# Create shortcut functions
function ai-save { cd C:\ALI_WORKSPACE\01_PROJECTS\ai-animation-factory; git add -A; git commit -m "Update: $(Get-Date -Format 'yyyy-MM-dd HH:mm')" }
function ai-push { cd C:\ALI_WORKSPACE\01_PROJECTS\ai-animation-factory; git push }
function ai-start { 
    docker start redis 2>$null; 
    Start-Process powershell -ArgumentList "-NoExit", "-Command", "cd C:\ALI_WORKSPACE\01_PROJECTS\ai-animation-factory\apps\api; pnpm dev"
    Start-Process powershell -ArgumentList "-NoExit", "-Command", "cd C:\ALI_WORKSPACE\01_PROJECTS\ai-animation-factory\apps\web; pnpm dev"
    Start-Sleep 5; Start-Process "http://localhost:3000"
}
function ai-stop { Get-Process node -ErrorAction SilentlyContinue | Stop-Process -Force; docker stop redis }

Write-Host "Shortcuts created:" -ForegroundColor Green
Write-Host "  ai-save  - Save changes" -ForegroundColor Gray
Write-Host "  ai-push  - Push to cloud" -ForegroundColor Gray
Write-Host "  ai-start - Start services" -ForegroundColor Gray
Write-Host "  ai-stop  - Stop services" -ForegroundColor Gray


| الأمر   | الوظيفة           | الاختصار       |
| ------- | ----------------- | -------------- |
| `save`  | حفظ التغييرات     | `git commit`   |
| `push`  | رفع للسحابة       | `git push`     |
| `start` | تشغيل الخدمات     | `pnpm dev`     |
| `stop`  | إيقاف الخدمات     | `Stop-Process` |
| `all`   | حفظ + رفع + تشغيل | الكل معاً      |
